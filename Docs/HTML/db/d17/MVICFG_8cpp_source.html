233 Graph_Line *findMatchedLine(Graph_Line *t, Graph *matchTo, Graph *matchFrom, Diff_Mapping diff) {
234  /* Extra check to ensure correct diff File */
235  if (diff.getFileName() != t->getGraphFunction()->getFunctionFile()) {
236  std::cerr << "findMatchedLine is using wrong diff File\n";
237  std::cerr << "Skipping match for " << t->getLineNumber(matchFrom->getGraphVersion()) << " from "
238  << matchFrom->getGraphVersion() << " to " << matchTo->getGraphVersion() << "\n";
239  return NULL;
240  } // End check for diff File name
241  unsigned lineFrom = t->getLineNumber(matchFrom->getGraphVersion());
242  unsigned lineTo = 0;
243  /* Check for virtual node */
244  if (matchFrom->isVirtualNodeLineNumber(lineFrom)) {
245  /* Virtual nodes have same unique line number across graphs*/
246  lineTo = lineFrom;
247  } else {
248  if (matchTo->getGraphVersion() > matchFrom->getGraphVersion()) {
249  /* Matching MVICFG to ICFG */
250  if (lineFrom == 0) {
251  /* New lines will have line numbers already in them */
252  lineTo = t->getLineNumber(matchTo->getGraphVersion());
253  } else {
254  lineTo = diff.getAfterLineNumber(lineFrom);
255  } // End check for lineFrom
256  } else {
257  /* Matching ICFG to MVICFG */
258  lineTo = diff.getBeforeLineNumber(lineFrom);
259  if (lineTo == 0) {
260  /* Line was deleted, but match is requested. Set lineTo to max
261  * and let the calling function deal with it */
262  lineTo = std::numeric_limits<unsigned>::max();
263  } // End check for lineTo
264  } // End check for matchTo > matchFrom
265  } // End check for isVirtualNodeLineNumber
266  if (lineTo != std::numeric_limits<unsigned>::max()) {
267  for (auto func : matchTo->getGraphFunctions()) {
268  if (func->getFunctionFile() == t->getGraphFunction()->getFunctionFile()) {
269  if (func->getFunctionName() == t->getGraphFunction()->getFunctionName()) {
270  std::list<Graph_Line *> matchedLines;
271  for (auto line : func->getFunctionLines()) {
272  if (lineTo == line->getLineNumber(matchTo->getGraphVersion())) {
273  std::string lineToString = getGraphLineInstructionsAsString(line);
274  std::string lineFromString = getGraphLineInstructionsAsString(t);
275  if (lineToString == lineFromString) {
276  return line;
277  } else {
278  matchedLines.push_back(line);
279  } // End check for lineTo and lineFrom string
280  } // End check for matching line number
281  } // End loop for lines
282  /* If there is a match at this point heuristically match it rather than return NULL */
283  if (!matchedLines.empty()) {
284  /* If only one match is there, then we don't have to work much */
285  if (matchedLines.size() == 1) {
286  return matchedLines.front();
287  } else {
288  std::string lineFromString = getGraphLineInstructionsAsString(t);
289  return resolveMatchedLinesWithNoExactStringMatch(matchedLines, lineFromString,
290  matchTo->getGraphVersion());
291  } // End check for matchedLines size
292  } // End check for empty matchedLines
293  } // End check for Function name check
294  } // End check for File name
295  } // End loop for functions
296  } // End check for virtual node check
297  return NULL;
298 } // End findMatchedLine

407  std::list<Graph_Line *> succ = getSuccGivenGraphLine(nDash);

413  if (findT == icfgN.end()) {

415  /* Proceed only if the function is in diff File being processed */
416  if (t->getGraphFunction()->getFunctionFile() == fileName) {
417  Graph_Line *tDash = findMatchedLine(t, MVICFG, ICFG, diff);

561  for (auto line : deletedLines) {

574  } // End loop for processing deletedGraphLines
575  } // End loop for identifying the deleted lines
576  for (auto func : MVICFG->getGraphFunctions()) {
577  /* Proceed only if the function is in diff File being processed */
578  if (func->getFunctionFile() == fileName) {
579  for (auto n : func->getFunctionLines()) {
580  auto findLine = std::find_if(std::begin(N), std::end(N), [=](Graph_Line *N) { return (N == n); });
581  if (findLine == N.end()) {

590  if (findT != N.end()) {

598  continue;

611  if (!m) {

614  } // End check for m

617  [=](Graph_Line *N) { return (N == mDash); });
618  auto findMSucc = std::find_if(std::begin(succDash), std::end(succDash),
619  [=](Graph_Line *N) { return (N == mDash); });
620  if (findMPred != predDash.end()) {

622  Graph_Instruction *mInst = m->getLineInstructions().back();
623  Graph_Instruction *nInst = n->getLineInstructions().front();
624  Graph_Edge *checkEdge = getEdge(mInst, nInst, Graph_Edge::ANY);

628  Graph_Instruction *mDashInst = mDash->getLineInstructions().front();

645  } // End check for foundEdge

647  Graph_Edge *newEdge = new Graph_Edge(mInst, nInst, edgeType, Version);
648  MVICFG->addEdge(mInst, nInst, newEdge);
649  } // End check for checkEdge
650  } else if (findMSucc != succDash.end()) {
651  /* Check for edge between n and m */
652  Graph_Instruction *nInst = n->getLineInstructions().back();
653  Graph_Instruction *mInst = m->getLineInstructions().front();
654  Graph_Edge *checkEdge = getEdge(nInst, mInst, Graph_Edge::ANY);

659  Graph_Edge *getEdgeType = getEdge(nDashInst, mDashInst, Graph_Edge::ANY);
660  Graph_Edge::edgeTypes edgeType;

666  if (checkBetweenEdge) {

669  } // End check for checkBetweenEdge

674  edgeType = Graph_Edge::MVICFG_DEL;

684  } // End check for m
685  } // End check to see if the function is in the same diff file

703 std::map<Graph_Line *, Graph_Line *> matchedInMVICFG(Graph *MVICFG, Graph *ICFG, Diff_Mapping diff, unsigned Version) {
704  std::map<long long, long long> matchedLines = diff.getMatchedLines();
705  std::string fileName = diff.getFileName();
706  std::map<Graph_Line *, Graph_Line *> matchedGraphLines;
707  std::list<Graph_Line *> mvicfgM;
708  /* Identify all the matched lines */
709  for (auto line : matchedLines) {
710  long long mvicfgLineNo = line.first;
711  long long icfgLineNo = line.second;
712  std::list<Graph_Line *> mvicfgGraphLines;
713  mvicfgGraphLines = getGraphLinesGivenLine(MVICFG, mvicfgLineNo, fileName);
714  /* No need to worry about matching lines that are not in MVICFG */
715  if (!mvicfgGraphLines.empty()) {
716  std::list<Graph_Line *> icfgGraphLines;
717  icfgGraphLines = getGraphLinesGivenLine(ICFG, icfgLineNo, fileName);
718  if (icfgGraphLines.empty()) {
719  std::cerr << "Graph_Line for line " << icfgLineNo << " not found in ICFG\n";
720  std::cerr << "Skipping this line and continuing\n";
721  continue;
722  } // End check for if line is present in ICFG
723  if (mvicfgGraphLines.size() != icfgGraphLines.size()) {
724  std::cerr << "Mismatch between the number of MVICFG and ICFG Graph_Lines for (" << line.first
725  << "::" << line.second << ")\n";
726  std::cerr << "Skipping this line and continuing\n";
727  continue;
728  } // End check for mismatch in GraphLine size
729  for (auto mvicfgLine = mvicfgGraphLines.begin(), icfgLine = icfgGraphLines.begin();
730  mvicfgLine != mvicfgGraphLines.end() && icfgLine != icfgGraphLines.end(); ++mvicfgLine, ++icfgLine) {
731  matchedGraphLines.insert(std::pair<Graph_Line *, Graph_Line *>(*icfgLine, *mvicfgLine));
732  mvicfgM.push_back(*icfgLine);
733  } // End loop for processing mvicfgGraphLines & icfgGraphLines
734  } // End check for if line is preset in MVICFG
735  } // End loop for matchedLines
736  for (auto line : matchedGraphLines) {
737  Graph_Line *n = line.second;
738  Graph_Line *nDash = line.first;
739  std::list<Graph_Line *> pred = getPredGivenGraphLine(nDash);
740  std::list<Graph_Line *> succ = getSuccGivenGraphLine(nDash);
741  std::list<Graph_Line *> T;
742  T.insert(T.end(), pred.begin(), pred.end());
743  T.insert(T.end(), succ.begin(), succ.end());
744  for (auto t : T) {
745  auto findT = std::find_if(std::begin(mvicfgM), std::end(mvicfgM), [=](Graph_Line *N) { return (N == t); });
746  if (findT != mvicfgM.end()) {
747  /* t in T and in Matched */
748  /* Proceed only if the function is in diff File being processed */
749  if (t->getGraphFunction()->getFunctionFile() == fileName) {
750  Graph_Line *tDash = findMatchedLine(t, MVICFG, ICFG, diff);
751  if (tDash) {
752  auto findTPred = std::find_if(std::begin(pred), std::end(pred), [=](Graph_Line *N) { return (N == t); });
753  auto findTSucc = std::find_if(std::begin(succ), std::end(succ), [=](Graph_Line *N) { return (N == t); });
754  if (findTPred != pred.end()) {
755  Graph_Instruction *nInst = n->getLineInstructions().front();
756  Graph_Instruction *tDashInst = tDash->getLineInstructions().back();
757  /* Edge should exist in the MVICFG. Raise error otherwise */
758  Graph_Edge *checkEdge = getEdge(tDashInst, nInst, Graph_Edge::ANY);
759  bool foundEdge = false;
760  if (!checkEdge) {
761  Graph_Edge *checkBetweenEdge = getInBetweenEdge(tDash, n);
762  if (checkBetweenEdge) {
763  foundEdge = true;
764  checkEdge = checkBetweenEdge;
765  } // End checkBetweenEdge
766  std::cerr << "MVICFG edge between " << tDashInst->getInstructionLabel() << " and "
767  << nInst->getInstructionLabel() << " not found\n";
768  std::cerr << "Skipping this predecessor edge\n";
769  } else {
770  foundEdge = true;
771  } // End check for checkEdge
772  if (foundEdge) {
773  if (!checkEdge->isPartOfGraph(Version)) {
774  checkEdge->pushEdgeVersions(Version);
775  } // End check for isPartOfGraph
776  } // End check for foundEdge
777  } else if (findTSucc != succ.end()) {
778  Graph_Instruction *tDashInst = tDash->getLineInstructions().front();
779  Graph_Instruction *nInst = n->getLineInstructions().back();
780  /* Edge should exist in the MVICFG. Raise error otherwise */
781  Graph_Edge *checkEdge = getEdge(nInst, tDashInst, Graph_Edge::ANY);
782  bool foundEdge = false;
783  if (!checkEdge) {
784  Graph_Edge *checkBetweenEdge = getInBetweenEdge(n, tDash);
785  if (checkBetweenEdge) {
786  foundEdge = true;
787  checkEdge = checkBetweenEdge;
788  } // End check for checkBetweenEdge
789  if (!foundEdge) {
790  std::cerr << "MVICFG edge between " << tDashInst->getInstructionLabel() << " and "
791  << nInst->getInstructionLabel() << " not found\n";
792  std::cerr << "Skipping this successor edge\n";
793  } // End check for foundEdge
794  } else {
795  foundEdge = true;
796  } // End check for checkEdge
797  if (foundEdge) {
798  if (!checkEdge->isPartOfGraph(Version)) {
799  checkEdge->pushEdgeVersions(Version);
800  } // End check for isPartOfGraph
801  } // End check for foundEdge
802  } // End check for Successor
803  } else {
804  std::cerr << "No matching line found for " << t->getLineNumber(ICFG->getGraphVersion())
805  << " in MVICFG(M)\n";
806  } // End check for tDash
807  } // End check to see if the function is in the same diff file
808  } // End check for find T
809  } // End loop for T
810  } // End loop for adding edges for matched lines
811  return matchedGraphLines;
812 } // End matchedInMVICFG

814 void updateMVICFGVersion(Graph *MVICFG, std::list<Graph_Line *> addedLines, std::list<Graph_Line *> deletedLines,
815  std::list<Diff_Mapping> diffMap, unsigned Version) {
816  /* Update Graph_Line information */
817  for (auto func : MVICFG->getGraphFunctions()) {
818  auto findDiff = std::find_if(std::begin(diffMap), std::end(diffMap),
819  [=](Diff_Mapping d) { return (d.getFileName() == func->getFunctionFile()); });
820  if (findDiff != diffMap.end()) {
821  for (auto line : func->getFunctionLines()) {
822  auto findInAdd = std::find_if(std::begin(addedLines), std::end(addedLines),
823  [=](Graph_Line *addLine) { return (addLine == line); });
824  auto findInDel = std::find_if(std::begin(deletedLines), std::end(deletedLines),
825  [=](Graph_Line *delLine) { return (delLine == line); });
826  if (findInAdd == addedLines.end() && findInDel == deletedLines.end()) {
827  /* Line was neither added nor deleted */
828  unsigned oldLineNumber = line->getLineNumber(MVICFG->getGraphVersion());
829  if (!MVICFG->isVirtualNodeLineNumber(oldLineNumber)) {
830  /* Substitute if the line is present in the MVICFG. Otherwise, it's mapping was done earlier */
831  if (oldLineNumber != 0) {
832  unsigned newLineNumber = findDiff->getAfterLineNumber(oldLineNumber);
833  if (newLineNumber != std::numeric_limits<unsigned>::max()) {
834  line->setLineNumber(Version, newLineNumber);
835  } else {
836  std::cerr << "Incorrect update line for " << oldLineNumber << "\n";
837  } // End check for newLineNumber being max
838  } // End check for oldLineNumber
839  } else {
840  /* Assign the same number in this version also */
841  unsigned newLineNumber = oldLineNumber;
842  line->setLineNumber(Version, newLineNumber);
843  } // End check for isVirtualNodeLineNumber
844  } // End check for deleted and added line
845  } // End loop for line
846  } else {
847  if (func->getFunctionFile() == "External_Node_File") {
848  /* Assign the same number in this version also */
849  for (auto line : func->getFunctionLines()) {
850  unsigned oldLineNumber = line->getLineNumber(MVICFG->getGraphVersion());

852  line->setLineNumber(Version, newLineNumber);
853  } // End loop for line
854  continue;
855  } // End check for External Node
856  std::cerr << "No Line mapping found for " << func->getFunctionFile() << "\n";
857  } // End check for diffMap.end
858  } // End loop for updating Graph_Line information
859  /* Collect all nodes from which a new edge has originated for this version */
860  std::list<Graph_Instruction *> mvicfgAddEdgesNodes;
861  for (auto mvicfgEdge : MVICFG->getGraphEdges()) {
862  Graph_Line *toLine = mvicfgEdge->getEdgeTo()->getGraphLine();

866  if (findInAddToLine != addedLines.end()) {
867  mvicfgAddEdgesNodes.push_back(mvicfgEdge->getEdgeFrom());
868  } // End check for addedLines.end
869  } // End loop for collecting Graph_Instruction
870  for (auto edge : MVICFG->getGraphEdges()) {
871  Graph_Instruction *edgeFromInst = edge->getEdgeFrom();
872  if (edgeFromInst->getGraphLine()->getLineNumber(Version) != 0) {
873  /* The from Node is active for this version */
874  Graph_Instruction *edgeToInst = edge->getEdgeTo();
875  if (edgeToInst->getGraphLine()->getLineNumber(Version) != 0) {
876  /* The to Node is active for this version */
877  auto findInAddEgdeFrom =
878  std::find_if(std::begin(mvicfgAddEdgesNodes), std::end(mvicfgAddEdgesNodes),
879  [=](Graph_Instruction *instComp) { return (edge->getEdgeFrom() == instComp); });
880  if (findInAddEgdeFrom == mvicfgAddEdgesNodes.end()) {
881  /* Neither the From node or To node were part of added edges */
882  if (!edge->isPartOfGraph(Version)) {
883  edge->pushEdgeVersions(Version);
884  } // End check for isPartOfGraph
885  } // End if for findInAddEgdeFrom
886  } // End check for edgeToInst
887  } // End check for edgeFromInst
888  } // End loop for updating Graph_Edge information
889 } // End updateMVICFGVersion
